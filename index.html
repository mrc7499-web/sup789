<script>

const API_URL="https://script.google.com/macros/s/AKfycbxHezmPZRW5x-PIEx7iPK5joGxshgIG_aN7wF0NvAl0FkIS0fgqnEKOI8oYR1RZN5cf/exec";


let DATA=[];

let currentCat="ทั้งหมด";

let editIndex=null;


/* ===== COLOR MAP (จัดหมวดหมู่สี) ===== */

const COLOR_MAP = {

  "ทั้งหมด":"blue",

  "สมัคร":"blue",

  "ขอข้อมูล / เปลี่ยนรหัส":"blue",

  "เปลี่ยนบัญชี":"blue",

  "ปิดแชท":"blue",

  "ทางเข้า":"blue",

  "ค้างเดิมพัน / เครดิต":"blue",


  "ฝากเงิน":"green",

  "QR Code":"green",


  "ถอนเงิน":"purple",


  "เครดิตฟรี / โปร":"pink",

  "เกม / แนะนำ":"pink",

  "กำลังใจ (เต็ม)":"pink",

  "ทักหลายไลน์":"pink",


  "ฝากปัญหา":"red",

  "บัญชีไม่ตรง":"red",

  "ถอนปัญหา":"red",

  "เข้าเล่นไม่ได้":"red",

  "VPN / เน็ต":"red",

  "ลูกค้าด่า":"red"

};


function getColor(cat){

  if(COLOR_MAP[cat]) return COLOR_MAP[cat];

  // Fallback

  for(const key in COLOR_MAP){

    if(cat.includes(key)){ return COLOR_MAP[key]; }

  }

  return "blue";

}


/* ===== INIT LOAD ===== */

async function init(){

  try {

    const res=await fetch(API_URL+"?t="+Date.now());

    DATA=await res.json();

    render();

  } catch (error) {

    console.error("Error loading data:", error);

    document.getElementById("grid").innerHTML = `<div class="loader" style="color:red;">❌ ไม่สามารถเชื่อมต่อฐานข้อมูลได้ กรุณารีเฟรชหน้าเว็บ</div>`;

  }

}


/* ===== TOAST & COPY ===== */

function showToast(msg){

  const t=document.getElementById("toast");

  t.innerText=msg;

  t.style.opacity=1;

  setTimeout(()=>t.style.opacity=0,1500);

}


async function copyText(text){

  try {

    await navigator.clipboard.writeText(text);

    showToast("✅ คัดลอกข้อความแล้ว");

  } catch (err) {

    console.error('Failed to copy: ', err);

    // Fallback

    const textArea = document.createElement("textarea");

    textArea.value = text;

    document.body.appendChild(textArea);

    textArea.select();

    document.execCommand("copy");

    textArea.remove();

    showToast("✅ คัดลอกข้อความแล้ว");

  }

}


async function copyImage(url){

  try {

    const res=await fetch(url);

    const blob=await res.blob();

    await navigator.clipboard.write([new ClipboardItem({[blob.type]:blob})]);

    showToast("🖼️ คัดลอกรูปภาพแล้ว");

  } catch (err) {

    console.error('Failed to copy image: ', err);

    showToast("❌ ไม่สามารถคัดลอกรูปได้อัตโนมัติ");

  }

}


/* ===== RENDER FUNCTION ===== */

function render(){

  const s=search.value.toLowerCase();

  let catsSet=new Set(["ทั้งหมด"]);


  // 1. โครงสร้างกลุ่ม

  const groups = {

    blue: { title: "💳 บัญชี ข้อมูล และ ทางเข้า", cards: [] },

    green: { title: "💰 การเงิน ฝากเงิน", cards: [] },

    purple: { title: "🔄 การถอนเงิน", cards: [] },

    pink: { title: "💖 ทั่วไป โปรโมท และ กำลังใจ", cards: [] },

    red: { title: "⚠️ ปัญหาขัดข้อง และ ลูกค้าคอมเพลน", cards: [] }

  };


  // 2. จัดการข้อมูล

  DATA.forEach((r,i)=>{

    const [cat,text,status,img]=r;

    if(status==="deleted") return;


    catsSet.add(cat);

    

    // กรอง

    if(currentCat!=="ทั้งหมด" && cat!==currentCat) return;

    if(s && !text.toLowerCase().includes(s)) return;


    const color = getColor(cat);


    // HTML การ์ด

    const cardHTML = `

    <div class="card ${color}" onclick="copyText(\`${text.replace(/`/g, '\\`')}\`)">

      <div class="actions">

        <div class="btn-action" onclick="event.stopPropagation();edit(${i})">✏️ แก้ไข</div>

        <div class="btn-action" onclick="event.stopPropagation();removeItem(${i})">🗑️ ลบ</div>

      </div>

      ${img ? `<img src="${img}" class="img-box" onclick="event.stopPropagation();copyImage('${img}')">` : ""}

      <div class="content">${text.replace(/{name}/g,'<span class="var">{name}</span>')}</div>

    </div>`;


    if(groups[color]) { groups[color].cards.push(cardHTML); }

  });


  // 3. วาด Grid และ Section

  let html="";

  Object.keys(groups).forEach(key => {

    const g = groups[key];

    if(g.cards.length > 0) { 

      html += `

      <div class="group-section">

        <div class="group-header">${g.title} <span style="font-size:12px; color:#888; font-weight:normal;">(${g.cards.length})</span></div>

        <div class="card-grid">

          ${g.cards.join("")}

        </div>

      </div>

      `;

    }

  });


  if(html==="") html="<div style='text-align:center; padding:50px; color:#888; font-size:16px;'>🔍 ไม่พบข้อความที่คุณค้นหา</div>";

  document.getElementById("grid").innerHTML=html;


  // 4. วาด Sidebar Menu

  let cHTML="";

  catsSet.forEach(c=>{

    cHTML+=`

      <div class="cat ${getColor(c)} ${c===currentCat?"active":""}" 

      onclick="currentCat='${c}';render()">

      ${c}

      </div>

    `;

  });

  document.getElementById("cats").innerHTML=cHTML;

}


/* ===== MODAL & CRUD ===== */

function openModal(){ 

  editIndex=null; 

  document.getElementById('catInput').value = currentCat !== "ทั้งหมด" ? currentCat : "";

  document.getElementById('textInput').value = "";

  document.getElementById('imgInput').value = "";

  document.getElementById('modal').style.display="flex"; 

}


function closeModal(){ document.getElementById('modal').style.display="none"; }


function edit(i){

  editIndex=i;

  document.getElementById('catInput').value=DATA[i][0];

  document.getElementById('textInput').value=DATA[i][1];

  document.getElementById('imgInput').value = ""; 

  document.getElementById('modal').style.display="flex";

}


async function removeItem(i){

  if(confirm("🗑️ คุณแน่ใจหรือไม่ว่าต้องการลบข้อความนี้? (การลบจะไม่สามารถกู้คืนได้)")){

    document.getElementById("grid").innerHTML = `<div class="loader">กำลังลบข้อมูล...</div>`;

    await fetch(API_URL,{

      method:"POST",

      body:JSON.stringify({action:"delete",index:i})

    });

    init();

  }

}


function save(){

  const imgInput = document.getElementById('imgInput');

  const file=imgInput.files[0];

  if(file){

    const reader=new FileReader();

    reader.onload=e=>executeSave(e.target.result);

    reader.readAsDataURL(file);

  }else{

    executeSave(null);

  }

}


async function executeSave(img){

  const catVal = document.getElementById('catInput').value.trim();

  const textVal = document.getElementById('textInput').value.trim();

  

  if(!catVal || !textVal) {

    alert("⚠️ กรุณาระบุหมวดหมู่และข้อความให้ครบถ้วน");

    return;

  }


  document.getElementById('modal').style.display="none"; 

  document.getElementById("grid").innerHTML = `<div class="loader">กำลังบันทึกข้อมูล...</div>`;


  await fetch(API_URL,{

    method:"POST",

    body:JSON.stringify({

      action:editIndex!==null?"edit":"add",

      index:editIndex,

      cat:catVal,

      text:textVal,

      image:img

    })

  });


  init();

}


/* ===== THEME ===== */

function toggleDark(){

  document.body.classList.toggle("dark");

  const isDark = document.body.classList.contains("dark");

  localStorage.setItem("zap789_theme", isDark ? "dark" : "light");

}


if(localStorage.getItem("zap789_theme") === "dark"){

  document.body.classList.add("dark");

}
